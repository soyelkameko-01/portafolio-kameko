
import React, { useState, useEffect } from 'react';
import { GALLERY_DATA } from '../constants';
import { GalleryItem } from '../types';
import AnimatedCover from './AnimatedCover';

const GalleryView: React.FC = () => {
  const [items, setItems] = useState<GalleryItem[]>(() => {
    const saved = localStorage.getItem('kameko_galleries');
    return saved ? JSON.parse(saved) : GALLERY_DATA;
  });
  const [selected, setSelected] = useState<GalleryItem | null>(null);
  const [showAddForm, setShowAddForm] = useState(false);
  
  // Estado para el formulario de nueva galería
  const [newGallery, setNewGallery] = useState({
    title: '',
    character: '',
    coverImage: '',
    description: ''
  });

  useEffect(() => {
    localStorage.setItem('kameko_galleries', JSON.stringify(items));
  }, [items]);

  const handleAddGallery = (e: React.FormEvent) => {
    e.preventDefault();
    if (!newGallery.title || !newGallery.coverImage) return;

    const newItem: GalleryItem = {
      id: Date.now().toString(),
      title: newGallery.title,
      character: newGallery.character,
      coverImage: newGallery.coverImage,
      images: [newGallery.coverImage], // Por defecto la misma que la portada
      description: newGallery.description || 'Nueva sesión añadida al archivo.'
    };

    setItems([newItem, ...items]);
    setNewGallery({ title: '', character: '', coverImage: '', description: '' });
    setShowAddForm(false);
  };

  if (selected) {
    return (
      <div className="min-h-screen pt-32 px-6 md:px-20 pb-20 animate-in fade-in duration-700 bg-black">
        <button 
          onClick={() => setSelected(null)} 
          className="mb-10 text-[10px] font-bold uppercase tracking-[0.3em] text-pink-500 hover:text-white transition-colors flex items-center gap-2"
        >
          <span>←</span> VOLVER AL ARCHIVO
        </button>
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-20">
          <div>
            <h2 className="text-5xl md:text-8xl font-black uppercase tracking-tighter mb-8 leading-none">
              {selected.title}
            </h2>
            <p className="text-neutral-500 text-lg leading-relaxed mb-10 max-w-xl">
              {selected.description}
            </p>
            <div className="inline-block px-4 py-2 bg-white/5 border border-white/10 rounded-full text-[10px] font-bold uppercase tracking-widest text-neutral-400">
              Personaje: {selected.character}
            </div>
          </div>
          <div className="space-y-8">
            {selected.images.map((img, i) => (
              <div key={i} className="overflow-hidden rounded-3xl bg-neutral-900 aspect-video md:aspect-auto">
                <img src={img} className="w-full h-full object-cover grayscale hover:grayscale-0 transition-all duration-1000" />
              </div>
            ))}
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen pt-20 px-6 md:px-20 pb-32">
      <div className="flex flex-col md:flex-row justify-between items-start md:items-end mb-20 gap-8">
        <div>
          <h1 className="text-6xl md:text-9xl font-black tracking-tighter leading-none uppercase">
            Archivo<br/><span className="text-pink-500">Visual</span>
          </h1>
        </div>
        
        <div className="flex flex-col items-end gap-4">
          <p className="text-[10px] text-neutral-500 font-bold uppercase tracking-[0.5em] text-right">
            Explora las sesiones más recientes
          </p>
          <button 
            onClick={() => setShowAddForm(!showAddForm)}
            className="px-6 py-3 bg-pink-500 text-white rounded-full text-[9px] font-bold uppercase tracking-widest hover:bg-white hover:text-black transition-all shadow-[0_0_20px_rgba(255,0,85,0.3)]"
          >
            {showAddForm ? 'Cerrar Editor' : '+ Añadir Nueva Sesión'}
          </button>
        </div>
      </div>

      {showAddForm && (
        <div className="mb-20 glass p-8 md:p-12 rounded-[2rem] border-pink-500/20 max-w-4xl mx-auto animate-in slide-in-from-top-4 duration-500">
          <form onSubmit={handleAddGallery} className="grid grid-cols-1 md:grid-cols-2 gap-6">
            <div className="space-y-2">
              <label className="text-[9px] font-bold uppercase tracking-widest text-neutral-500">Título de la Sesión</label>
              <input 
                value={newGallery.title}
                onChange={e => setNewGallery({...newGallery, title: e.target.value})}
                placeholder="Ej: Blood Moon" 
                className="w-full bg-black/40 border border-white/10 rounded-xl px-4 py-3 text-sm focus:border-pink-500 transition-all outline-none"
              />
            </div>
            <div className="space-y-2">
              <label className="text-[9px] font-bold uppercase tracking-widest text-neutral-500">Personaje / Serie</label>
              <input 
                value={newGallery.character}
                onChange={e => setNewGallery({...newGallery, character: e.target.value})}
                placeholder="Ej: Akali - KDA" 
                className="w-full bg-black/40 border border-white/10 rounded-xl px-4 py-3 text-sm focus:border-pink-500 transition-all outline-none"
              />
            </div>
            <div className="md:col-span-2 space-y-2">
              <label className="text-[9px] font-bold uppercase tracking-widest text-neutral-500">URL de la Imagen de Portada</label>
              <input 
                value={newGallery.coverImage}
                onChange={e => setNewGallery({...newGallery, coverImage: e.target.value})}
                placeholder="Pega el enlace de la imagen aquí..." 
                className="w-full bg-black/40 border border-white/10 rounded-xl px-4 py-3 text-sm focus:border-pink-500 transition-all outline-none"
              />
            </div>
            <div className="md:col-span-2 space-y-2">
              <label className="text-[9px] font-bold uppercase tracking-widest text-neutral-500">Breve Descripción</label>
              <textarea 
                value={newGallery.description}
                onChange={e => setNewGallery({...newGallery, description: e.target.value})}
                placeholder="Cuéntanos algo sobre esta sesión..." 
                className="w-full bg-black/40 border border-white/10 rounded-xl px-4 py-3 text-sm focus:border-pink-500 transition-all outline-none h-24 resize-none"
              />
            </div>
            <button 
              type="submit"
              className="md:col-span-2 py-4 bg-white text-black font-black uppercase tracking-[0.2em] text-[10px] rounded-xl hover:bg-pink-500 hover:text-white transition-all"
            >
              Publicar en el Archivo
            </button>
          </form>
        </div>
      )}

      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-8">
        {items.map(item => (
          <AnimatedCover 
            key={item.id} 
            image={item.coverImage} 
            title={item.title} 
            subtitle={item.character} 
            onClick={() => setSelected(item)} 
          />
        ))}
      </div>
    </div>
  );
};

export default GalleryView;
